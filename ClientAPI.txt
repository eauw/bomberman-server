# Befehle von Client zu Server

Player geht ein Feld nach oben:			"move up","w"

Player geht ein Feld nach unten:		"move down","s"

Player geht ein Feld nach links:		"move left","a"

Player geht ein Feld nach rechts:		"move right","d"

Player wirft Bombe:				 		"bomb","b"

Player beendet seine Züge für diese Runde: "n"

Game State: "l"

# Befehle von Server an Client:
"yt" => Wenn der Spieler dran ist.

"nyt" => Wenn der Spieler nicht dran ist.
